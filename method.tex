\section{Setup}
\subsection{Problems}
Consider a problem of $K(K\geq1)$ stage dynamic treatment regimes estimation by maximizing quantile value with time-varying unmeasured confounders under the proximal causal inference framework.

\subsection{Notations}
\begin{itemize}
    \item $Y_k$: outcome variables
        \begin{itemize}
            \item $k=0$: the baseline measurement of the outcome
            \item $k=1,\ldots,K-1$: the intermediate outcome
            \item $k=K$: the ultimate outcome
        \end{itemize}
    \item $A_k$: dichotomous treatment variables satisfy $A_k \in \mathcal{A}_k=\mathcal{A}_k, \, \forall k=1,2,\ldots,K$
    \item $U_k$: time-varying unmeasured confounders, $k=0,1,\ldots,K-1$
    \item $Z_k$: treatment-inducing proxies, $k=1,2,\ldots,K$
    \item $W_k$: outcome-inducing proxies, $k=1,2,\ldots,K$
    \item $\bar{X}_k$: the sequence from the initial stage to $k$th stage of variable $X$
    \item $\mathcal{D}_k$: the class of candidate regimes at $k$th stage, and $\mathcal{D} =\{\mathcal{D}_1,\ldots, \mathcal{D}_K\}$ denotes the set of all candidate regime sequences
    \item $d_k$: the regime function $d_k=d_k(\bar{Y}_{k-1},\bar{a}_{k-1})$, where $d_k \in \mathcal{D}_k$ 
    \textcolor{blue}{\begin{rem}: $\bar{Y}_{k-1}$ also depends on $\bar{a}_{k-1}$, thus $d_k$ can also be written as $d_k=d_k(\bar{Y}_{k-1}(\bar{a}_{k-1}),\bar{a}_{k-1})$\end{rem}}
\end{itemize}

\subsection{Identification Framework}
\begin{itemize}
    \item Potential outcomes for $K$ stages: define $Y_k(\bar{d}_k)$ as the intermediate potential outcomes at (until) stage $k$. Then the ultimate potential outcomes $Y_{K}(\bar{d}_K)$ that would be achieved under regime $\bar{d}_K=(d_1,d_2,\ldots,d_K)$ are collected through the sequence
    \[
    \{Y_0, Y_1(d_1),Y_2(\bar{d}_2),\ldots,Y_{K-1}(\bar{d}_{K-1}),Y_K(d_K)\},
    \]
    where we define
    \[
    Y_k(\bar{d}_k) = \sum_{\bar{a}_k}[Y_k(\bar{a}_k)\prod_{j=1}^k I(d_j(\bar{Y}_{j-1}(\bar{a}_{j-1}),\bar{a}_{j-1}) = a_j)], \quad k=1,2,\ldots,K.
    \]
    \item Value function: the value function for regime $\bar{d}_K$ defined as 
    \[
    V(\bar{d}_K) = \mathbb{E}[Y_K(\bar{d}_K)]
    \]
    \item Quantile-value function: similar to the value function, we denote the quantile-value function as 
    \[
    QV(\tau,\bar{d}_K) = Q_{\tau}[Y_K(\bar{d}_K)],
    \]
    where $\tau \in (0,1)$ and $Q_{\tau}(\bar{d}_K)$ denotes the $\tau$th quantile of $\bar{d}_K$. Specifically, $Q_\tau[Y_K(\bar{d}_K)] = \inf \{t: F_d(\cdot)>\tau\}$, where $F_d$ is the distribution function of $Y_K(\bar{d}_K)$.
    \item Goal: find the quantile-optimal regimes $d^*=\bar{d}_K^*$ that maximize the quantile-value function
    \[
    d^* = \bar{d}_K^*= \argmax_{d\in\mathcal{D}} Q_\tau[Y_K(\bar{d}_K)].
    \]
    \item Characterization of $\tau$th quantile: for a given treatment regime$d=\bar{d}_K$, an estimation of quantile-value is 
    \[\begin{aligned}
    QV(\tau,\bar{d}_K) &= \argmin_{q} \mathbb{E}[\rho_{\tau}(Y_K(\bar d_K)-q)] \\
    &= \argmin_{q} \sum\limits_{\bar y_K}\sum\limits_{\bar a_K}[\rho_{\tau}(y_K-q)\prod_{j=1}^K I\left(d_j(\bar y_{j-1}, \bar a_{j-1})=a_j\right)f(\bar Y_K(\bar a_K)=\bar y_K)],
    \end{aligned}\]
    where $\rho_\tau(u) := u(\tau-I\left(u<0\right))$ is the check function (quantile loss).
    
    Then we can characterize the quantile-optimal DTR as 
    \[\begin{aligned}
        d^* = \argmax_{d\in \mathcal{D}}\biggl\{ \argmin_{q} \sum\limits_{\bar y_K}\sum\limits_{\bar a_K}[\rho_{\tau}(y_K-q) 
        \prod_{j=1}^K I\left(d_j(\bar y_{j-1}, \bar a_{j-1})=a_j\right)
        f(\bar Y_K(\bar a_K)=\bar y_K \mid Y_0=y_0) f(Y_0=y_0)]  \biggr\}.
    \end{aligned}\]
    Thus, it is critical to identify the conditional joint density of the potential outcomes:
    \[f(\bar Y_K(\bar a_K)=\bar y_K \mid Y_0=y_0).\]
\end{itemize}

\subsection{Identifiability Assumptions}
\begin{ass}[Sequential consistency]
    $Y_k = Y_k(\bar A_k)$ for $k=1,2,\ldots,K$ almost surely.
    \label{ass10}
\end{ass}

\begin{ass}[Treatment-inducing confounding proxies]
    $Y_k(\bar a_k, \bar z_k) = Y_k(\bar a_k)$ for $k=1,2,\ldots,K$ almost surely
    $\Longleftrightarrow Y_k \pperp \bar Z_k \mid (\bar U_{k-1}, \bar Y_{k-1})$.
    \label{ass12}
\end{ass}

\begin{ass}[Sequential proximal latent randomization]
    $\{Y_k(\bar a_k), Y_{k+1}(\bar a_{k+1}), \ldots, Y_K(\bar a_K), \bar W_k\} \pperp \{A_k,\bar Z_k\}|(\bar U_{k-1}, \bar Y_{k-1}, \bar A_{k-1}=\bar a_{k-1})$ for $k=1,2,\ldots,K$ and $a_k \in \mathcal{A}_k$.
    \label{ass13}
\end{ass}

\begin{ass}[Sequential positivity]
    $0< f(A_k=a_k \mid \bar U_{k-1}, \bar Y_{k-1}, \bar A_{k-1}) < 1$ for $k=1,2,\ldots,K$ and $\forall a_k \in \mathcal{A}_k$.
    \label{ass14}
\end{ass}

\begin{ass}[Z-Completeness]
    For a fixed $k$ and $m=K,K-1,\ldots,k+1$, for any square integrable function $g_m(\bar U_{m-1})$ and $\forall \bar y_{m-1}, \bar a_m$,
    \[
    \mathbb{E}[g_m(\bar U_{m-1}) \mid \bar Z_m, \bar a_m, \bar y_{m-1}]=0, \,a.s. \Longleftrightarrow g_m(\bar U_{m-1}) = 0, \, a.s. .
    \]
    \label{ass15}
\end{ass}

\begin{ass}[W-Completeness]
    For a fixed $k$ and $n=1,2,\ldots,k$, for any square integrable function $g_n(\bar U_{n-1})$ and $\forall \bar y_{n-1}, \bar a_n$,
    \[
    \mathbb{E}[g_n(\bar U_{n-1}) \mid \bar W_n, \bar a_n, \bar y_{n-1}]=0, \,a.s. \Longleftrightarrow g_n(\bar U_{n-1}) = 0, \, a.s. .
    \]
    \label{ass16}
\end{ass}

\begin{ass}[Treatment confounding bridge functions]
    For a fixed $k$ and $n=1,2,\ldots,k$, there exist treatment confounding bridge functions $q_{nn}(\cdot)$ such that
    \begin{numcases}{}
        \frac{1}{f(a_1 \mid w_1,y_0)} = \sum_{z_1}q_{11}(z_1,y_0,a_1)f(z_1 \mid a_1,w_1,y_0), & $n=1$ \label{tbf1}\\
        \frac{\sum\limits_{\bar z_{n-1}} q_{n-1,n-1}(\bar z_{n-1}, \bar y_{n-2}, \bar a_{n-1})f(\bar z_{n-1} \mid \bar a_{n-1}, \bar w_n, \bar y_{n-1})}{f(a_n \mid \bar a_{n-1}, \bar w_n, \bar y_{n-1})} = \sum_{\bar z_n}q_{nn}(\bar z_{n}, \bar y_{n-1}, \bar a_{n})f(\bar z_n \mid \bar a_n, \bar w_n, \bar y_{n-1}), & $n=2,\ldots,k$. \label{tbf2}
    \end{numcases}
    \label{ass18}
\end{ass}


\subsection{Identification and estimation}
We first give an identification expression for $K$ stage.
\begin{thm}
    Suppose assumptions \ref{ass10}-\ref{ass18} holds. Then we have
    \begin{align}
        f(\bar Y_k(\bar a_K)=\bar y_K \mid Y_0=y_0) = \sum_{\bar z_K} q_{KK}(\bar z_K, 
        \bar y_{K-1}, \bar a_K)f(\bar z_K, \bar y_K, \bar a_K \mid y_0). 
    \end{align}
    \label{thm1}
\end{thm}

Now, for further testing and experimentation, we only consider the situation when $K=2$.
It is natural to obtain an estimate of the target, i.e., the quantile-value function, from the observed data.
\begin{align}
    &\hat{QV}(\tau,\bar d_K) \notag \\
    =& \argmin_{q}  \mathbb{P}_n \bigl\{\rho_\tau(Y_{2}-q)\cdot \hat q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(d_1(Y_{0})=A_1)I(d_2(\bar Y_{1},A_1)=A_2) \bigr\} \notag \\
    =& \argmin_{q} \frac{1}{n}\sum_{i=1}^n  \bigl\{\rho_\tau(Y_{2,i} - q)\cdot \hat q_{22}(\bar Z_{2,i}, \bar Y_{1,i}, \bar A_{2,i}) I(d_1(Y_{0,i})=A_{1,i})I(d_2(\bar Y_{1,i},A_{1,i})=A_{2,i}) \bigr\}.
    \label{est}
\end{align}



\section{Policy Learning}
Now that we have already derived how to use the treatment bridge function to estimate quantile-value function from\eqref{est}, the next step is to learn the quantile-optimal regimes (policies) $\bar d_2^* = (d_1^*, d_2^*)$ in the policy space $\mathcal{D}$ that maximize the quantile-value function, which can be identified as
\begin{align}
    \bar d_2^* = (d_1^*, d_2^*) =&  \argmax_{(d_1,d_2) \in \mathcal{D}} Q_\tau(Y_2(d_1, d_2))    \notag\\
     =& \argmax_{(d_1,d_2) \in \mathcal{D}} \Big\{\argmin_{q}\mathbb{E} \bigl[\rho_\tau(Y_{2}-q)\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(d_1(Y_{0})=A_1)I(d_2(\bar Y_{1},A_1)=A_2) \bigr]\Big\}.  \label{ident}
\end{align}
Solving the quantile-optimal treatment regime $(d_1^*,d_2^*)$ can be deemed as a bi-level optimization with the following two-level optimization:
\begin{itemize}
    \item Inner level optimization: for a fixed $d=(d_1,d_2)$, first solve the quantile-value function to find the optimal quantile:
    \begin{align}
    & Q_\tau(Y_2(d_1,d_2)) = \argmin\limits_{q} \mathbb{E}\big[\rho_\tau(Y_{2}(d_1,d_2) - q) \big] \label{inner}
    \end{align}
    \item Outer level optimization: fixing $Q_\tau(Y_2(d_1,d_2))$, then find the quantile-optimal treatment regimes  
    \begin{align}
    d^* = (d_1^*,d_2^*) = \argmax \limits_{(d_1,d_2)} Q_\tau(Y_2(d_1,d_2)).
    \label{outer}
    \end{align}
\end{itemize}

For the inner level optimization in \eqref{inner}, the quantile-value function can be identified via PIPW \eqref{est} by 
\begin{align}
    &Q_\tau(Y_2(d_1,d_2)) = \argmin_{q}\mathbb{E} \bigl\{\rho_\tau(Y_{2}-q)\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(d_1(Y_{0})=A_1)I(d_2(\bar Y_{1},A_1)=A_2) \bigr\} \notag\\
    \Longleftrightarrow \quad & \mathbb{E}\big[I\{Y_2 > Q_\tau(Y_2(d_1,d_2))\}\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(d_1(Y_{0})=A_1)I(d_2(\bar Y_{1},A_1)=A_2) \big]= 1-\tau. \label{ilo1}
\end{align}
The above derivation indicates that if we find the optimal quantile value $q^*$ that satisfies the equation \eqref{ilo1} under the fixed regimes $d_1,d_2$, i.e., 
\begin{align}
    \mathbb{E}\big[I\{Y_2 > q^*\}\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(d_1(Y_{0})=A_1)I(d_2(\bar Y_{1},A_1)=A_2) \big]= 1-\tau, \label{ilo2}
\end{align}
the optimal $q^*$ can be regarded as a function of polices $d_1,d_2$.
In other words, equation \eqref{ilo2} establishs an implicit function of $q^*$ and $d_1,d_2$,
which allows us to consider $q^*$ as a function of $d_1,d_2$, denoted as $q^*(d_1,d_2)$.

Consequently, the outer level optimization can be simplified as solving
\begin{align}
    (d_1^{*},d_2^{*}) = \argmax_{(d_1,d_2)\in\mathcal{D}}q^*(d_1,d_2). \notag
\end{align}
Furthermore, we consider $q$ as a parameter since our goal is only to find the optimal $d_1,d_2$. 
Denote the survival value function as
$$
SV(d_1,d_2;q) = \mathbb{E} \big[I(Y_2 > q)\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \cdot  I(d_1(Y_0)=A_1)I(d_2(\bar Y_1, A_1)=A_2) \big].
$$
The goal of outer level optimization can be expressed as 
\begin{align}
    (\hat d_1^*,\hat d_2^*) = \argmax_{(d_1,d_2)\in\mathcal{D}} SV(d_1,d_2;q^*).
\end{align}

In practice, we plan our policy learning in the following steps.
First, we need to pre-estimate the nuisance function $\hat q_{22}(\bar Z_2,\bar Y_1, \bar A_2)$.
Then, we solve the inner level optimization and outer level optimization via alternating optimization.


\subsection{Pre-estimation of the treatment bridge function}
Before estimating the policies, we first need to estimate the nuisance treatment bridge function $q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})$, which also requires the estimation of $q_{11}$ according to assumption \ref{ass18}.
Our method involves using neural networks to estimate the treatment bridge functions, which builds on the idea of the DeepMMR method proposed by (Zhang2025).
Furthermore, we extended their method to a two-stage setting for estimating the treatment bridge function $q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})$.

Recalling the recursive definition of the treatment bridge function in assumption \ref{ass18}, we have the following conditional moment restrictions for $q_{11}$ and $q_{22}$:
\begin{align}
    & \mathbb{E}\big[I(A_1=a_1)q_{11}(Z_1, Y_0, A_1)-1|W_1,Y_0\big] = 0, \\
    & \mathbb{E}\big[I(A_2=a_2)q_{22}(\bar Z_2, \bar Y_1, \bar A_2)- q_{11}(Z_1, Y_0, A_1)|A_1, \bar W_2,\bar Y_1\big] = 0.
\end{align}
Based on this view, we extend our method.
Let $\mathcal{P}_{k_1}$ be a reproducing kernel Hilbert space (RKHS) defined on $(\mathcal{W}_1, \mathcal{Y}_0)$ with a reproducing kernel $k_1:(\mathcal{W}_1, \mathcal{Y}_0)\times (\mathcal{W}_1, \mathcal{Y}_0) \to \mathbb{R}$ that satisfies
(a) $\forall (w_1,y_0)\in (\mathcal{W}_1, \mathcal{Y}_0), k_1((w_1,y_0),(\cdot,\cdot)) \in \mathcal{P}_{k_1}$, and 
(b) $\forall (w_1,y_0)\in(\mathcal{W}_1, \mathcal{Y}_0),p_1\in\mathcal{P}_{k_1},  p_1(w_1,y_0)= \langle p_1,k_1((w_1,y_0),(\cdot,\cdot))\rangle_{k_1}$,
where $\langle\cdot,\cdot\rangle_{k_1}$ denotes the inner product of $\mathcal{P}_{k_1}$. 
Similarly, let $\mathcal{P}_{k_2}$ be a reproducing kernel Hilbert space (RKHS) defined on $(\mathcal{A}_1,\mathcal{W}_1,\mathcal{W}_2, \mathcal{Y}_0, \mathcal{Y}_1)$ with a reproducing kernel $k_2:(\mathcal{A}_1,\mathcal{W}_1,\mathcal{W}_2, \mathcal{Y}_0, \mathcal{Y}_1)\times (\mathcal{A}_1,\mathcal{W}_1,\mathcal{W}_2, \mathcal{Y}_0, \mathcal{Y}_1) \to \mathbb{R}$ that satisfies the properties similar to (a) and (b). We can also denote $p_2(a_1,\bar w_2,\bar y_1)$ as the function on $\mathcal{P}_{k_2}$, and $\langle\cdot,\cdot\rangle_{k_2}$ as the inner product of $\mathcal{P}_{k_2}$.
Specifically, we require $k_1$ and $k_2$ to be continuous, bounded, and integrally strictly positive definite (ISPD).
Then, we can define the risk of estimating $q_{11}$ and $q_{22}$ as
\begin{align}
    R_1(q_{11}):= R^{a_1}_{k_1}(q_{11})=& \mathbb{E}\big\{(I(A_1=a_1)q_{11}(Z_1,Y_0,A_1)-1) (I(A_1'=a_1')q_{11}(Z_1',Y_0',A_1')-1)\times k_1((W_1,Y_0),(W_1',Y_0'))  \big\}, \\
    R_2(q_{22}):= R^{a_2}_{k_2}(q_{22})=& \mathbb{E}\big\{(I(A_2=a_2)q_{22}(\bar Z_2,\bar Y_1,\bar A_2)-q_{11}(Z_1,Y_0,A_1)) \times \notag\\
    & (I(A_2'=a_2')q_{22}(\bar Z_2',\bar Y_1',\bar A_2')-q_{11}(Z_1',Y_0',A_1'))\times k_2((A_1, \bar W_2,\bar Y_1),(A_1', \bar W_2',\bar Y_1'))  \big\},
\end{align}
where $(\bar A_2', \bar W_2',\bar Z_2', \bar Y_2')$ is an independent copy of the random variables $(\bar A_2, \bar W_2,\bar Z_2, \bar Y_2)$.
According to the risk minimization criteria, we can estimate $q_{11}$ and $q_{22}$ by minimizing the empirical risks of U-statistic:
\begin{align}
    & \hat q_{11}=\argmin_{q_{11}} \frac{1}{n(n-1)}\sum_{i,j=1, i\neq j}^n (I(A_{1,i}=a_{1,i})q_{11,i}-1)(I(A_{1,j}=a_{1,j})q_{11,j}-1)\cdot k_{1,ij} + \lambda_1||q_{11}||_2^2, \label{q11hat}\\
    & \hat q_{22}=\argmin_{q_{22}} \frac{1}{n(n-1)}\sum_{i,j=1, i \neq j}^n (I(A_{2,i}=a_{1,i})q_{22,i}-\hat q_{11,i})(I(A_{2,j}=a_{2,j})q_{22,j}-\hat q_{11,j})\cdot k_{2,ij} + \lambda_2||q_{22}||_2^2, \label{q22hat}
\end{align}
where $q_{tt,i}=q_{tt}(\bar Z_{t,i}, \bar Y_{t-1,i}, \bar A_{t,i})$,  $k_{t,ij}=((\bar A_{t-1,i}, \bar W_{t,i},\bar Y_{t-1,i}),(\bar A_{t-1,j}, \bar W_{t,j},\bar Y_{t-1,j}))$ , and $\lambda_1, \lambda_2$ are the tuning parameters that control the $L_2-$penalty terms for $t=1,2$.
In practice, we use neural networks to estimate the first $\hat q_{11}$ and then $\hat q_{22}$.



\subsection{Inner Level Optimization}
For fixed $d_1,d_2$, solving equation \eqref{ilo1} can be deemed as minimizing the following loss function with respect to $q$:
\begin{align}
    L(q;d_1,d_2) = \Big| SV(d_1,d_2;q) - (1-\tau) \Big|.
\end{align}
In fact, it is actually a one-dimensional search problem.
Since the nuisance estimator $\hat q_{22}$ may be negative so that the target function is non-monotonic, we adopt grid search method to solve the optimization problem.
Initially, we divide the search interval $[l, u]$ into $N$ discrete points to form a grid
$\mathcal{Q} = \{q_1,\cdots,q_N\}$.
Note that the indicator $I(Y_{2,i}>q)$ in the loss function changes its value only when $q=Y_{2,i}$, which allows us simply sort the outcome variables and set $N=n$ and $\mathcal{Q} = \{Y_{2,(1)},\cdots,Y_{2,(n)}\}$.
Then we calculate the value of loss function in each point $q_i \in \mathcal{Q}$.
The global optimal quantile value $q^*(d_1,d_2)$ under the regimes $d_1,d_2$ is obtained by 
\begin{align}
    q^*(d_1,d_2) = \argmin_{q_i \in \mathcal{Q}} L(q;d_1,d2).
\end{align}

In addition, it can be easily shown that the grid search method can converge to the global optimum according to Lipschitz continuity bound.


\subsection{Outer level optimization}
Motivated by (\textbf{Laha2021}), we consider solving the original optimization problem as a weighted classification problem using a surrogate loss to smooth the indicator function of $d$. 
For a fixed $q$, consider an alternative form of the targeted survival function 
\begin{align}
    &SV(f_1,f_2;q) = \mathbb{E} \big[I(Y_2 > q)\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \cdot  I(A_1f_1(H_1)>0)I(A_2f_2(H_2)>0 \big] \\
    \Longleftrightarrow\quad & SV(d_1,d_2;q) = \mathbb{E} \big[I(Y_2 > q)\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \cdot  I(d_1(H_1)=A_1)I(d_2(H_2)=A_2) \big],
\end{align}
where $H_1=\{Y_0\}, H_2=\{\bar Y_1(A_1)\} \subset \{\bar Y_1,A_1\}:= H_2^\dag$.
\begin{rem}[]
    It is essential to notice that the historical information set $H_t$ relies on the potential outcomes of previous stages under the proximal setting (Y.Gao, J.Zhang), which is different from the setting of the no unmeasured confounder assumption (NUCA).
    However, it is not intuitive to be the direct input parameters of the treatment regime $d_t$. 
    In fact, the set $H_t=\{\bar Y_{t-1}(\bar A_{t-1})\}$ can be viewed as a random variable that contains the necessary information yielded by observed data $H_t^\dag =\{\bar Y_{t-1}, \bar A_{t-1}\}$. 
    The notation $d_t(H_t)$ implies the treatment regime $d_t$ is actually determined by the observed outcomes of previous stages $\bar Y_{t-1}$ and the practical treatments $\bar a_{t-1}$ which lead to these outcomes.
    To prevent confusion, from the second stage, we suggest to use $H_t$ to denote the variables associated with treatment regime $d_t(H_t)$ (e.g., decision variables $f_t(H_t)$, $SQ_t(H_t; A_t,q),\eta_t(H_t)$ that would be introduced later), and $H_t^\dag$ to be other variables (e.g., condition set).
\end{rem}

Similar to (\textbf{\textbf{Laha2021}}), we can further define a form with a surrogate loss of the targeted survival function as
\begin{align}
    SV_\psi(f_1,f_2;q) = \mathbb{E} \big[I(Y_2 > q)\cdot q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \cdot  \psi(A_1f_1(H_1),A_2f_2(H_2)) \big],
\end{align}
where $\psi(x,y) = \phi(x)\phi(y)$ is the surrogate loss that introduced in (\textbf{Laha2021}).
Also denote the ($\psi-$)optimal targeted survival function as
\begin{align*}
    & SV^*=SV(f_1^*,f_2^*;q)=\sup\limits_{f_1,f_2 \in \mathcal{F}}SV(f_1,f_2;q),  \\
    & SV_\psi^*= SV_\psi(\tilde f_1, \tilde f_2;q) = \sup\limits_{f_1,f_2 \in \mathcal{F}}SV_\psi(f_1,f_2;q),
\end{align*}
where $\mathcal{F}=\Big\{(f_1,f_2)|f_1:\mathcal{H}_1\to\mathbb{R}, \quad f_2:\mathcal{H}_2\to\mathbb{R} \;\text{are measurable}\Big\}$, and $\tilde f_1,\tilde f_2$ are extended valued functions that can take the value of $\pm\infty$ suggested by (\textbf{Laha2021}).
Then it is direct to characterize $d_1^*(H_1)=sign(f_1^*(H_1)), d_2^*(H_2)=sign(f_2^*(H_2))$ and $\tilde d_1(H_1)=sign(\tilde f_1(H_1)), \tilde d_2(H_2)=sign(\tilde f_2(H_2))$.
There are some specific choice of surrogate loss $\psi(x,y) = \phi(x)\phi(y)$ offered by (Laha2021):
\begin{itemize}
    \item $\phi(x)=1+ \frac{2}{\pi}\arctan{(\frac{\pi x}{2})}$;
    \item $\phi(x) = 1 + \frac{x}{1+|x|}$;
    \item $\phi(x) = 1+\frac{x}{\sqrt{1+x^2}}$;
    \item $\phi(x) = 1+ \tanh(x)$.
\end{itemize}
In practical, due to the restriction of function class $\mathcal{F}$, we actually optimize the target $\hat{SV}_\psi(f_1,f_2;q)$ over a nested class $\mathcal{U}_{1}\subset \cdots \subset\mathcal{U}_{n}\subset\mathcal{F}$, where $\mathcal{U}_{t}=(\mathcal{U}_{1t},\mathcal{U}_{2t})$ such that $f_{1} \in \mathcal{U}_{1t}$ and $f_{2} \in \mathcal{U}_{2t}$ for $t=1,\dots,n$.
We also introduce a regularity term with $L_2$-penalty to avoid overfitting, which is controlled by the hyperparameter $\lambda_n$.

Under these settings, if the optimal quantile value $q^*$ is found, the learning goal of outer level optimization is to find $(\hat f_{n,1}, \hat f_{n,2})$ such that 
\begin{align}
        (\hat f_{n,1},\hat f_{n,2})
        = \argmax_{(f_1,f_2) \in \mathcal{U}_n}
        \Big\{\frac{1}{n} \sum_{i=1}^n  \big[I(Y_{2,i}>q^*) \cdot \hat q_{22}(\bar Z_{2,i}, \bar Y_{1,i}, \bar A_{2,i}) \psi(A_{1,i}f_1(H_{1,i}),A_{2,i}f_2(H_{2,i})) \big] - \lambda_n(||f_1||_{n,2}^2+||f_2||_{n,2}^2) \Big\}
        \label{ole1}
\end{align}
and the solution leads to the estimated regimes $\hat d_{n,1} = \text{sign}(\hat f_{n,1})$ and $\hat d_{n,2} = \text{sign}(\hat f_{n,2})$.





\subsection{Learning Algorithm}
\begin{algorithm}[htb]
    \caption{Quantile-Optimal DTR Estimation via Alternating Optimization}
    \SetKwInOut{Input}{Input}
    \SetKwInOut{Output}{Output}
    \SetKwIF{If}{ElseIf}{Else}{if}{:}{elif}{else:}{}

    \Input{Observed data $\{\bar Y_{2,i},\bar Z_{2,i},\bar A_{2,i}, \bar W_{2,i}\}^n_{i=1}$, quantile level $\tau$, grid set $\mathcal{Q}$ (e.g., sorted unique values of $Y_{2}$), max iterations $K_{max}$, tolerance thresholds $\varepsilon_n, \delta_n$, bandwidth $h_n$}
    \Output{Estimated optimal policies and optimal quantile value}
    
    \BlankLine
    Estimate nuisance treatment bridge function $\hat{q}_{22}(\bar Z_2,\bar Y_1, \bar A_2)$ by \eqref{q11hat} and \eqref{q22hat}\;
    Initialize surrogate quantile $q^{(0)}$ (e.g., sample $\tau$-th quantile of $Y_2$)\;
    \For{$k=1,2,\cdots, K_{max}$}
    {
        \textbf{Step 1. Policy Update (Outer level):} Fix $q^{(k-1)}$, solve for the optimal policies:
        \begin{flalign*}
            (\hat f_{n,1}^{(k)},\hat f_{n,2}^{(k)}) = &\argmax_{(f_1,f_2) \in \mathcal{U}_n}\frac{1}{n} \sum_{i=1}^n  \big[I(Y_{2,i}>q^{(k-1)}) \cdot  
            \hat q_{22,i} \cdot \psi(A_{1,i}f_1(H_{1,i}),A_{2,i}f_2(H_{2,i})) \big] - \lambda_n(||f_1||_{n,2}^2+||f_2||_{n,2}^2);
        \end{flalign*} \\
        \textbf{Step 2. Quantile Update (Inner level):} Fix current policies $(\hat f_{n,1}^{(k)},\hat f_{n,2}^{(k)})$, evaluate over the grid $\mathcal{Q}$ to find the new quantile:
        \begin{flalign*}
            q^{(k)} \leftarrow \argmin_{q \in \mathcal{Q}} \left| \hat{SV}_{\Phi}(\hat{f}_{n,1}^{(k)},\hat{f}_{n,2}^{(k)};q;{h_n}) - (1-\tau) \right|;
        \end{flalign*}\\
        \textbf{Step 3. Convergence Check:} \\
        \If{$|q^{(k)} - q^{(k-1)}| \leq \delta_n$ \textbf{\upshape or} $\left|\hat{SV}_{\Phi}(\hat{f}_{n,1}^{(k)},\hat{f}_{n,2}^{(k)};q^{(k)};{h_n}) - (1-\tau)\right| \leq \varepsilon_n$ \textbf{\upshape or} $\sum_{t=1}^2 \sum_{i=1}^n I\big( \text{sign}(\hat{f}_{n,t}^{(k)}(H_{t,i})) \neq \text{sign}(\hat{f}_{n,t}^{(k-1)}(H_{t,i})) \big) = 0$}{\textbf{break}\;}

        \BlankLine
    }
    $\hat{d}_{n,1} \leftarrow \text{sign}(\hat f_{n,1}^{(k)}), \: \hat{d}_{n,2} \leftarrow \text{sign}(\hat f_{n,2}^{(k)})$ \;
    \Return $\{(\hat{d}_{n,1},\hat{d}_{n,2}), m^{(k)}\}$\;
    \label{alg1}
\end{algorithm}

In this section, we summarize the complete alternating optimization process of our method, Prox\_qtr\_sl, for estimating the optimal policies $(\hat d_{1,n}, \hat d_{2,n})$ in Algorithm \ref{alg1}. 

To evaluate the estimated policies, we introduce an estimator (Jiang2017, Xia2025) of the survival function:
\begin{align}
    \hat{SV}_{\Phi}(\hat{f}_{n,1}^{(k)},\hat{f}_{n,2}^{(k)};m^{(k)};{h_n}) = \frac{1}{n} \sum_{i=1}^n  \big[I(Y_{2,i}>m^{(k)}) \cdot \hat q_{22}(\bar Z_{2,i}, \bar Y_{1,i}, \bar A_{2,i}) \Phi\{A_{1,i}\hat{f}_{n,1}^{(k)}(H_{1,i})/h_n\}\Phi\{A_{2,i}\hat{f}_{n,2}^{(k)}(H_{2,i})/h_n\} \big], \notag
\end{align}
where $\Phi(\cdot)$ denotes the distribution function of the standard normal distribution, and $h_n$ is the bandwidth hyperparameter.
It offers another soft surrogate form $\Phi\{A_{1}f_{1}/h_n\}$ and $\Phi\{A_{2}f_{2}/h_n\}$
for hard indicators of the policies, which can help to improve computational stability.
We also consider three conditions to detect convergence: (1) the estimator $\hat{SV}_{\Phi}(\hat{f}_{n,1}^{(k)},\hat{f}_{n,2}^{(k)};q;{h_n})$ is close enough to $(1-\tau)$, which is controlled by parameter $\varepsilon_n$; (2) the new quantile is close enough to the old, which is controled by parameter $\delta_n$; (3) the signals of estimaed policies (i.e., the estimated regimes) no longer change.
The chosen values of some parameters are as follows:
\begin{itemize}
    \item grid set $\mathcal{Q}$: the unique sorted distribution values of the outcome variable $\{Y_{2,(1)},\cdots,Y_{2,(n)}\}$;
    \item max iterations $K_{max}$: 20;
    \item bandwidth $h_n$:  $0.2/\log n$,
    \item convergence tolerance for survival value $\varepsilon_n$: $10^{-4}$,
    \item convergence tolerance for quantile threshold $\delta_n$: $10^{-4}$.
\end{itemize}




\section{Theoretical Results}
In this section, we aim to demonstrate the justification of using surrogate loss to replace the indicator function for policy learning.
To this end, we first prove the Fisher consistency.
Besides, we provide an excess risk bound under finite samples.

\subsection{Fisher consistency}
Before showing the Fisher consistency, a necessary trick is to map the optimal DTR problem as a weighted binary classification problem.
For this purpose, we first introduce another identification method of optimal treatment regime from (Gao2025). Denote
\begin{align}
    & SQ_2(h_2;a_2,q):= SQ_2(\bar y_1,a_1;a_2,q) = \mathbb{E}[I(Y_2(a_1,a_2)>q)|\bar Y_1(a_1)=\bar y_1], \\
    & SQ_1(h_1;a_1,q) := SQ_1(y_0;a_1,q) = \mathbb{E}[\max_{a_2}SQ_2(\bar y_1,a_1;a_2,q)|Y_0=y_0],
\end{align}
then
\begin{align}
    & d_2^*(h_2;q):= d_2^*(\bar y_1,a_1;q) = \argmax_{a_2} SQ_2(\bar y_1,a_1;a_2,q),\\
    & d_1^*(h_1;q):= d_1^*(y_0;q) = \argmax_{a_1} SQ_1(y_0;a_1,q).
\end{align}
It is obvious that $SQ_1, SQ_2 >0$, and under the same identifiable assumption in Theorem \ref{thm1}, we can further identify $SQ_1$ and $SQ_2$ as:
\begin{align}
    SQ_2(\bar y_1,a_1;a_2,q) =& \frac{\sum\limits_{y_2}I(y_2>q)\sum\limits_{\bar z_2}q_{22}(\bar z_2,\bar y_1,\bar a_2)f(\bar z_2,\bar y_2,\bar a_2|y_0)}{\sum\limits_{y_2}\sum\limits_{\bar z_2}q_{22}(\bar z_2,\bar y_1,\bar a_2)f(\bar z_2,\bar y_2,\bar a_2|y_0)} \notag\\ 
    =& \frac{\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_2, \bar Y_1, \bar A_2)|\bar Y_1= \bar y_1, \bar A_2= \bar a_2\big]}{\mathbb{E}\big[q_{22}(\bar Z_2, \bar Y_1, \bar A_2)|\bar Y_1= \bar y_1, \bar A_2= \bar a_2\big]}, \label{sq2}
\end{align}
\begin{align}
    SQ_1(y_0;a_1,q) =& \sum_{y_1}\Big\{\max_{a_2}SQ_2(\bar y_1,a_1;a_2,q) \times
     \frac{\sum\limits_{y_2}\sum\limits_{\bar z_2}q_{22}(\bar z_2,\bar y_1,\bar a_2)f(\bar z_2,\bar y_2,\bar a_2|y_0)}{\sum\limits_{y_2,y_1}\sum\limits_{\bar z_2}q_{22}(\bar z_2,\bar y_1,\bar a_2)f(\bar z_2,\bar y_2,\bar a_2|y_0)}\Big\} \notag\\
    =& \frac{\mathbb{E}\big[\big(\max\limits_{a_2}SQ_2(\bar y_1,a_1;a_2,q)\big)q_{11}(Z_1, Y_0, A_1)|Y_0= y_0, A_1= a_1\big]}{\mathbb{E}\big[q_{11}(Z_1, Y_0, A_1)|Y_0= y_0, A_1= a_1\big]} \label{sq1}
\end{align}

Based on this view, we can construct the following map.
For the second stage, define the map: $\eta_2: \mathcal{H}_2\to\mathbb{R}$,
\begin{align}
    \eta_2(h_2) = \frac{SQ_2(h_2;1,q)}{SQ_2(h_2;1,q)+SQ_2(h_2;-1,q)}.
    \label{eta2}
\end{align}
For the first stage, according to the backward induction of optimal DTR, define the map: $\eta_1: \mathcal{H}_1\to\mathbb{R}$,
\begin{align}
    \eta_1(h_1) = \frac{SQ_1(h_1;1,q)}{SQ_1(h_1;1,q)+SQ_1(h_1;-1,q)}.
    \label{eta1}
\end{align}
Then it follows that $d^*_t(H_t) = 1$ if $\eta_t(H_t)>1/2$ and $d^*_t(H_t) = -1$ if $\eta_t(H_t)<1/2$ for $t=1,2$, and the decision boundaries are decided by $\{h_1:\eta_1(h_1)=1/2\}$ and $\{h_2:\eta_1(h_2)=1/2\}$.

With the definition of $\eta_1(H_1)$ and $\eta_2(H_2)$, we now introduce a lemma derived from \textbf{Lemma H.1} from (\textbf{Laha2021}), which can also be proved according to the latter.
\begin{lem}[]
    Suppose assumptions \ref{ass10}-\ref{ass14}, \ref{ass16} and \ref{ass18} hold, and $\psi(x,y)=\phi(x)\phi(y)$ satisfies condition \ref{cond2laha}. Then for a fixed q, $(\tilde f_1,\tilde f_2) \in \bar{\mathcal{F}}$ satisfy $SV_\psi(\tilde f_1,\tilde f_2)=\sup\limits_{(f_1,f_2)\in \mathcal{F}} SV_\psi(f_1,f_2)$ if
    \[
    \tilde f_1(H_1) = \begin{cases}
        \infty & \eta_1(H_1) > 1/2\\
        -\infty & \eta_1(H_1) < 1/2
    \end{cases},    
    \quad \text{and} \quad
    \tilde f_2(H_2) = \begin{cases}
        \infty & \eta_2(H_2) > 1/2\\
        -\infty & \eta_2(H_2) < 1/2
    \end{cases}.
    \]
    \label{lemh1laha}
\end{lem}
The following lemma from \textbf{Lemma H.2} in \textbf{Laha2021} can be directly utilized in the following proof. 
\begin{lem}[\textbf{Lemma H.2} in \textbf{Laha2021}]
    Suppose $\phi$ satisfies Condition \ref{cond2laha}, and define $C_t(x) = t\phi(x)+(1-t)\phi(-x)$ for $t\in(0,1)$. Then $C_t(x)$ is maximized at $x=\infty$ if $t>1/2$ or at $x=-\infty$ if $t<1/2$, and is a constant on $\mathbb{R}$ if $t=1/2$. Moreover, for each $t\in[0,1]$,
    \begin{align*}
        & \sup_{x\in\mathbb R}C_t(x) = C_\phi\max\{t,1-t\}, \\
        & \sup_{x\in\mathbb R}C_t(x)-\sup_{x:x(2t-1)\leq0}C_t(x)=C_\phi|t-1/2|.
    \end{align*}
    \label{lemh2laha}
\end{lem}
The following lemma shows the essence of treatment bridge functions that are helpful in the following proof. 
It can be shown directly from assumption \ref{ass18}.
\begin{lem}[]
    Suppose assumption \ref{ass18} holds, then 
    \begin{align}
        &\mathbb{E}[q_{11}(Z_1,Y_0,A_1)|A_1,Y_0] = \frac{1}{P(A_1|Y_0)}, \\
        &\mathbb{E}[q_{22}(\bar Z_2,\bar Y_1,\bar A_2)|\bar A_2, \bar Y_1] = \frac{\mathbb{E}[q_{11}(Z_1,Y_0,A_1)|A_1,\bar Y_1]}{P(A_2|A_1,\bar Y_1)}.
    \end{align}
    \label{lemass18}
\end{lem}
\begin{proof}
    For the first equality, we have
    \begin{align}
        \mathbb{E}[q_{11}(Z_1,Y_0,A_1)|A_1,Y_0] =& \sum_{w_1} \mathbb{E}[q_{11}(Z_1,Y_0,A_1)|A_1,Y_0,W_1=w_1]f(w_1|A_1,Y_0) \notag\\
        =& \sum_{w_1} \frac{1}{f(A_1|W_1=w_1,Y_0)}f(w_1|A_1,Y_0) \notag\\
        =& \sum_{w_1} \frac{1}{f(A_1|W_1=w_1,Y_0)}\cdot\frac{f(A_1|W_1=w_1,Y_0)f(w_1|Y_0)}{f(A_1|Y_0)} = \frac{1}{P(A_1|Y_0)}.
    \end{align}
    For the second equality, we have
    \begin{align}
        \mathbb{E}[q_{22}(\bar Z_2,\bar Y_1,\bar A_2)|\bar A_2, \bar Y_1] =& \sum_{\bar w_2}\mathbb{E}[q_{22}(\bar Z_2,\bar Y_1,\bar A_2)|\bar A_2, \bar Y_1, \bar W_2=\bar w_2]f(\bar w_2|\bar A_2, \bar Y_1) \notag\\
        =& \sum_{\bar w_2}\frac{\mathbb{E}[q_{11}(Z_1, Y_0, A_1)|A_1, \bar Y_1, \bar W_2=\bar w_2]}{f(A_2|A_1,\bar Y_1,\bar W_2=\bar w_2)}f(\bar w_2|\bar A_2, \bar Y_1)\notag\\
        =& \sum_{\bar w_2}\frac{\mathbb{E}[q_{11}(Z_1, Y_0, A_1)|A_1, \bar Y_1, \bar W_2=\bar w_2]}{f(A_2|A_1,\bar Y_1,\bar W_2=\bar w_2)}\cdot \frac{f(A_2|A_1,\bar Y_1,\bar W_2=\bar w_2)f(\bar w_2|A_1,\bar Y_1)}{f(A_2|A_1,\bar Y_1)} \notag\\
        =&\frac{1}{P(A_2|A_1,\bar Y_1)}\cdot\sum_{\bar w_2}\mathbb{E}[q_{11}(Z_1, Y_0, A_1)|A_1, \bar Y_1, \bar W_2=\bar w_2]f(\bar w_2|A_1,\bar Y_1) \notag\\
        =&\frac{\mathbb{E}[q_{11}(Z_1, Y_0, A_1)|A_1, \bar Y_1]}{P(A_2|A_1,\bar Y_1)}.
    \end{align}
\end{proof}


And the following condition from (\textbf{Laha2021}) is helpful to construct a surrogate loss function: 
\begin{cond}[\textbf{Condition 2} in \textbf{Laha2021}]
        $\phi$ is a strictly increasing function such that
        \begin{enumerate}  
            \item $\forall x \in \mathbb{R}, \phi(x) > 0$.
            \item $\forall x \in \mathbb{R}, \phi(x)+\phi(-x)=C_\phi$, where $C_\phi >0$ is a constant.
            \item $\lim\limits_{x\to\infty} \phi(x)=C_\phi$, and $\lim\limits_{x\to-\infty}\phi(x)=0$.
        \end{enumerate}
        \label{cond2laha}
\end{cond}

\begin{thm}[Fisher consistency]
    Suppose assumptions \ref{ass10}-\ref{ass14}, \ref{ass16} and \ref{ass18} hold, then for a fixed q, \\
    (a) For any $\{f_{1n},f_{2n}\}_{n\>1} \in \mathcal{F}$ that satisfies  $SV_\psi(f_{1n}, f_{2n};q) \to SV_\psi^*$ also satisfies $SV(f_{1n},f_{2n};q) \rightarrow SV^*$; \\
    (b) (regret bound restriction) Suppose $\psi(x,y)=\phi(x)\phi(y)$ satisfies condition \ref{cond2laha}, then for the constant $C_\phi$ that satisfies condition \ref{cond2laha}, 
    \begin{align}
        SV^* - SV(f_1,f_2;q) \leq \frac{SV_\psi^* - SV_\psi(f_1, f_2;q)}{{(C_\phi^2/4)}}.
    \end{align}
    \label{fishercon}
\end{thm}

\begin{proof}
    It is straightforward to validate (a) as long as we prove (b).
    Therefore, we mainly show (b) based on (\textbf{Laha2021}). 
    For simplicity, denote $f_t:=f_t(H_t)$ for $t=1,2$ and $\tilde f_t, f_t^*$ is denoted in the same way.
    On one hand, we have 
    \begin{align}
        &SV_\psi^* - SV_\psi(f_1,f_2;q) \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) [\psi(A_1\tilde f_1, A_2\tilde f_2)) - \psi(A_1f_1, A_2f_2)]\big] \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) [\phi(A_1\tilde f_1)\phi(A_2\tilde f_2) - \phi(A_1f_1\phi(A_2f_2)]\big] \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) [\phi(A_1\tilde f_1)\phi(A_2\tilde f_2) - \phi(A_1 f_1)\phi(A_2\tilde f_2) + \phi(A_1 f_1)\phi(A_2\tilde f_2) \phi(A_1 f_1)\phi(A_2f_2)]\big] \notag \\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})\phi(A_1f_1)[\phi(A_2\tilde f_2)-\phi(A_2f_2)]\big] +
        \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})\phi(A_2\tilde f_2)[\phi(A_1\tilde f_1)-\phi(A_1f_1)]\big] \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})\phi(A_1f_1)[\phi(A_2\tilde f_2)-\phi(A_2f_2)]\big] +
        C_\phi\mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2d_2^*>0)[\phi(A_1\tilde f_1)-\phi(A_1f_1)]\big], \label{fcpf1}
    \end{align}
    where the last equality follows lemma \eqref{lemh1laha} since it is defined that $\phi(\infty)=C_\phi$ and $\phi(-\infty)=0$ and the fact that $sign(\tilde f_2)=\tilde d_2= d_2^*=sign(f_2^*)$. 
    
    The next step is to lower bound the twotermsm in \eqref{fcpf1}.
    For the first term, we have
    \begin{align}
        & \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})\phi(A_1f_1)\big(\phi(A_2\tilde f_2)-\phi(A_2f_2)\big)\big] \notag\\
        =& \mathbb{E} \Big[\phi(A_1f_1) \mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})(\phi(A_2\tilde f_2)-\phi(A_2f_2))| H_2^\dag \big]\Big] \notag\\
        =& \mathbb{E} \Big[\phi(A_1f_1) \mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})(\phi(A_2\tilde f_2)-\phi(A_2f_2))| \bar Y_1, A_1\big]\Big] \notag\\
        =& \mathbb{E} \Big[\phi(A_1f_1)\Big\{P(A_2=1|\bar Y_1,A_1)\cdot\mathbb{E}[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})|\bar Y_1,A_1,A_2=1]\cdot (\phi(\tilde f_2)-\phi(f_2)) \notag \\
        &+ P(A_2=-1|\bar Y_1,A_1)\cdot\mathbb{E}[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})|\bar Y_1,A_1,A_2=-1]\cdot (\phi(-\tilde f_2)-\phi(-f_2)) \Big\} \Big] \notag\\
        =& \mathbb{E} \Big[\phi(A_1f_1)R_2(H_2^\dag)\Big\{SQ_2(H_2;1,q)(\phi(\tilde f_2)-\phi(f_2)) + SQ_2(H_2;-1,q)(\phi(-\tilde f_2)-\phi(-f_2)) 
        \Big\}\Big],
    \end{align}
    where $R_2(H_2^\dag) = P(A_2=a_2|\bar Y_1,A_1)\cdot \mathbb{E}[q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})|\bar Y_1,A_1,A_2=a_2]
    =\mathbb{E}\big[q_{11}(Z_1,Y_0,A_1)|\bar Y_1,A_1\big]
    =\mathbb{E}\big[q_{11}(Z_1,Y_0,A_1)|H_2^\dag\big]$ for $a_2=\{1,-1\}$ by Lemma \ref{lemass18}.\\
    We then show that $R_2(H_2^\dag) > 0$. 
    In fact, by Lemma \ref{lemass18},
    \begin{align}
        R_2(h_2^\dag) &= \mathbb{E}\big[q_{11}(Z_1,Y_0,A_1)|h_2^\dag\big] \notag\\
        &= \sum_{z_1}q_{11}(z_1,y_0,a_1)f(z_1|\bar y_1,a_1) \notag\\
        &= \sum_{u_0}\sum_{z_1} q_{11}(z_1,y_0,a_1)f(z_1|\bar y_1,a_1,u_0)f(u_0|a_1,\bar y_1) \notag\\
        &= \sum_{u_0}\sum_{z_1} q_{11}(z_1,y_0,a_1)f(z_1|y_0,a_1,u_0)f(u_0|a_1,\bar y_1) \notag\\
        &= \sum_{u_0}\frac{f(u_0|a_1,\bar y_1)}{f(a_1|y_0, u_0)} \notag\\
        &= \sum_{u_0}\frac{1}{f(a_1|u_0,y_0)}\cdot \frac{f(y_1|a_1,y_0,u_0)f(a_1|y_0,u_0)f(u_0|y_0)}{f(y_1|a_1,y_0)f(a_1|y_0)} \notag\\
        &= \frac{1}{f(y_1,a_1|y_0)}\sum_{u_0}f(y_1|a_1,y_0,u_0)f(u_0|y_0) \notag\\
        &= \frac{f(Y_1(a_1)=y_1|y_0)}{f(y_1,a_1|y_0)} > 0,
    \end{align}
    where the fourth equality is by assumption \ref{ass12}, and the fifth equality is by the original assumption by treatment bridge function (see Cui2020), and the last equality is by the backdoor adjustment formula.

    
    Let $T_1=SQ_2(H_2;1,q)(\phi(\tilde f_2)-\phi(f_2)) + SQ_2(H_2;-1,q)(\phi(-\tilde f_2)-\phi(-f_2))$, by the definition of $\eta_2(H_2)$ in \eqref{eta2}, we have  
    \begin{align}
    T_1 &= \big(SQ_2(H_2;1,q) + SQ_2(H_2;-1,q)\big)\big(C_{\eta_2(H_2)}(\tilde f_2)-C_{\eta_2(H_2)}(f_2)\big) \notag\\
    &\geq \big(SQ_2(H_2;1,q) + SQ_2(H_2;-1,q)\big)\big(C_{\eta_2(H_2)}(\tilde f_2)-C_{\eta_2(H_2)}(f_2)\big)  I(H_2\in \mathcal{E}_2(f)) \notag \\
    &\geq \frac{C_\phi}{2}\big|SQ_2(H_2;1,q) - SQ_2(H_2;-1,q) \big| I(H_2\in \mathcal{E}_2(f)), \label{fcpft11}
    \end{align}
    where $\mathcal{E}_2(f)=\{h_2\in \mathcal H_2:sign(f_2(h_2)) \neq d^*_2(h_2)\}$.
    The last inequality is by Lemma \ref{lemh2laha} and the property of $\tilde f_2$ in lemma \ref{lemh1laha} that 
    \[
    C_{\eta_2(H_2)}(\tilde f_2) - C_{\eta_2(H_2)}(f_2) \geq C_\phi|\eta_2(H_2)-1/2|.
    \]
    Also note that for $H_2 \notin \mathcal{E}_2(f)$, $I(A_2\tilde f_2>0) - I(A_2f_2>0) = 0$ since $\tilde d_2 = d_2^*$ holds, then we have
    \begin{align}
            &\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)|H_2^\dag\big] \notag\\
            =& \mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)|H_2^\dag\big]I(H_2 \in \mathcal{E}_2(f)) \notag\\
            =& R_2(H_2^\dag)\Big\{SQ_2(H_2;1,q)(I(\tilde f_2>0)-I(f_2>0)) + SQ_2(H_2;-1,q)(I(\tilde f_2<0)-I(f_2<0)) \Big\} I(H_2 \in \mathcal{E}_2(f)) \notag\\
            =& R_2(H_2^\dag)\big|SQ_2(H_2;1,q) - SQ_2(H_2;-1,q) \big| I(H_2\in \mathcal{E}_2(f)), \label{fcpft12}
        \end{align}
    where the last equality follows the fact that $f_2(H_2)>0$ implies $\tilde f_2(H_2)<0$ over the set $H_2 \in \mathcal{E}_2(f)$.\\
    By \eqref{fcpft11} and \eqref{fcpft12}, we can lower bound the first term in \eqref{fcpf1} that
    \begin{align}
        \text{First Term} =& \mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})\phi(A_1f_1)[\phi(A_2\tilde f_2)-\phi(A_2f_2)]\Big] \notag\\
        \geq& \frac{C_\phi}{2}\mathbb{E} \Big[\phi(A_1f_1)R_2(H_2^\dag) \big|SQ_2(H_2;1,q) - SQ_2(H_2;-1,q) \big| I(H_2\in \mathcal{E}_2(f))  \Big] \notag\\
        =& \frac{C_\phi}{2}\mathbb{E}\Big[\phi(A_1f_1)\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)|H_2^\dag\big] \Big] \notag\\
        \geq& \frac{C_\phi}{2}\mathbb{E}\Big[\phi(A_1f_1)\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big) \cdot I(A_1f_1>0)|H_2^\dag\big] \Big] \notag\\
        \geq& \frac{C_\phi^2}{4}\mathbb{E}\Big[\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big) \cdot I(A_1f_1>0)|H_2^\dag\big] \Big] \notag\\
        =& \frac{C_\phi^2}{4}\mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(A_1f_1>0) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)\Big], \label{firstterm}
    \end{align}
    where the last equality is given by condition \ref{cond2laha} that if $x>0$, then $\phi(x)>\phi(0)=C_\phi/2$.
    
    For the second term in \eqref{fcpf1}, we can lower bound it in a quite similar way with that of the first term. 
    Note that 
    \begin{align}
        &\mathbb{E}\Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2d_2^*>0)\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big)|H_1\Big] \notag\\
        =& \mathbb{E}\Big[\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2d_2^*>0)|\bar Y_1,A_1\big]\big|Y_0\Big]\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big) \notag\\
        =& \mathbb{E}\Big[\Big\{P(A_2=1|\bar Y_1,A_1)\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})|\bar Y_1,A_1,A_2=1\big] \notag\\
        &+ P(A_2=-1|\bar Y_1,A_1)\mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})|\bar Y_1,A_1,A_2=-1\big]\Big\}I(A_2=d_2^*)\big|Y_0\Big]\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big) \notag\\
        =& \mathbb{E}\Big[R_2(H_2)\big[SQ_2(H_2;1,q)+SQ_2(H_2;-1,q)\big]I(A_2=d_2^*) \big|Y_0\Big]\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big) \notag\\
        =& \mathbb{E}\Big[\mathbb{E}\big[q_{11}(Z_1,Y_0,A_1)|\bar Y_1,A_1\big]SQ_2^*(H_2;d_2^*,q)\big|Y_0\Big]\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big) \notag\\
        =& \mathbb{E}\Big[SQ_2^*(H_2;d_2^*,q)q_{11}(Z_1,Y_0,A_1)\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big)|Y_0\Big] \notag\\
        =& P(A_1=1|Y_0)\mathbb{E}\Big[SQ_2^*(H_2;d_2^*,q)q_{11}(Z_1,Y_0,A_1)|Y_0,A_1=1\Big]\big(\phi(\tilde f_1)-\phi(f_1)\big) \notag\\
        &+ P(A_1=-1|Y_0)\mathbb{E}\Big[SQ_2^*(H_2;d_2^*,q)q_{11}(Z_1,Y_0,A_1)|Y_0,A_1=-1\Big]\big(\phi(-\tilde f_1)-\phi(-f_1)\big) \notag\\
        =& SQ_1(H_1;1,q)\big(\phi(\tilde f_1)-\phi(f_1)\big) + SQ_1(H_1;-1,q)\big(\phi(-\tilde f_1)-\phi(-f_1)\big),
    \end{align}
    where $SQ_2^*(h_2,d_2^*;q) = \max\limits_{a_2}SQ_2(h_2;a_2,q)$ by definition,
    and the last equality follows \eqref{sq1} by assumption \ref{ass18} that 
    \begin{align*}
        SQ_1(h_1;a_1,q) = SQ_1(y_0;a_1,q) &= \frac{\mathbb{E}\big[\big(\max\limits_{a_2}SQ_2(\bar y_1,a_1;a_2,q)\big)q_{11}(Z_1, Y_0, A_1)|Y_0= y_0, A_1= a_1\big]}{\mathbb{E}\big[q_{11}(Z_1, Y_0, A_1)|Y_0= y_0, A_1= a_1\big]} \notag\\
        &= \mathbb{E}\big[SQ_2^*(\bar y_1,a_1;d_2^*,q)q_{11}(Z_1, Y_0, A_1)|Y_0= y_0, A_1= a_1\big]P(A_1=a_1|Y_0=y_0).
    \end{align*}
    Let $T_2 = SQ_1(H_1;1,q)\big(\phi(\tilde f_1)-\phi(f_1)\big) + SQ_1(H_1;-1,q)\big(\phi(-\tilde f_1)-\phi(-f_1)\big)$ similar to $T_1$, then we have
    \begin{align}
        T_2 &= \big(SQ_1(H_1;1,q)+SQ_1(H_1;-1,q)\big)\big(C_{\eta_1(H_1)}(\tilde f_1)-C_{\eta_1(H_1)}(f_1)\big) \notag\\
        &\geq \big(SQ_1(H_1;1,q)+SQ_1(H_1;-1,q)\big)\big(C_{\eta_1(H_1)}(\tilde f_1)-C_{\eta_1(H_1)}(f_1)\big)I(H_1\in\mathcal{E}_1(f)) \notag\\
        &\geq \frac{C_\phi}{2}|SQ_1(H_1;1,q)-SQ_2(H_1;-1,q)|I(H_1\in \mathcal{E}_2(f)), \label{fcpft21}
    \end{align}
    where $\mathcal{E}_1(f)=\{h_1\in \mathcal H_1:sign(f_1(h_1)) \neq d^*_1(h_1)\}$, and the last inequality is also by Lemma \ref{lemh2laha}.\\
    Also note that for $H_1 \notin \mathcal{E}_1(f)$, $I(A_1\tilde f_1>0)-I(A_1f_1>0)=0$ since $\tilde d_1 = d_1^*$ holds, then we have
    \begin{align}
        & \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2\tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)|H_1\big] \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2 \tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)|H_1\big] I(H_1\in \mathcal{E}_1) \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2 d_2^*>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)|H_1\big] I(H_1\in \mathcal{E}_1) \notag\\
        =& \Big\{SQ_1(H_1;1,q)\big(I(\tilde f_1>0)-I(f_1>0)\big)+SQ_1(H_1;-1,q)\big(I(\tilde f_1<0)-I(f_1<0)\big)\Big\}I(H_1\in \mathcal{E}_2(f)) \notag\\
        =& |SQ_1(H_1;1,q)-SQ_2(H_1;-1,q)|I(H_1\in \mathcal{E}_2(f)), \label{fcpft22}
    \end{align}
    where the last equality follows the fact that $f_1(H_1)>0$ implies $\tilde f_1(H_1)<0$ over the set $H_1\in \mathcal{E}_2(f))$.\\
    By \eqref{fcpft21} and \eqref{fcpft22}, we can also lower bound the second term in \eqref{fcpf1} that
    \begin{align}
        \text{Second Term} =& C_\phi\mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2d_2^*>0)\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big)\Big] \notag\\
        =& C_\phi\mathbb{E}\Big[ \mathbb{E}\big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2d_2^*>0)\big(\phi(A_1\tilde f_1)-\phi(A_1f_1)\big)|H_1\big]\Big] \notag\\
        =& C_\phi\mathbb{E}\Big[SQ_1(H_1;1,q)\big(\phi(\tilde f_1)-\phi(f_1)\big) + SQ_1(H_1;-1,q)\big(\phi(-\tilde f_1)-\phi(-f_1)\big)  \Big] \notag\\
        \geq& \frac{C^2_\phi}{2}\mathbb{E}\Big[|SQ_1(H_1;1,q)-SQ_2(H_1;-1,q)|I(H_1\in \mathcal{E}_2(f))\Big] \notag\\
        =& \frac{C^2_\phi}{2}\mathbb{E} \Big[\mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2\tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)|H_1\big]\Big] \notag\\
        =& \frac{C^2_\phi}{2}\mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2\tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)\Big]. \label{secondterm}
    \end{align}
    
    On the other hand, since $sign(\tilde f_t)=sign(f_t^*)$ for $t=1,2$, we have
    \begin{align}
        &SV^*-SV(f_1,f_2;q)  \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_1f_1>0)\big(I(A_2f_2^*>0)-I(A_2f_2>0)\big)\big] \notag\\
        &+ \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2 f_2^*>0)\big(I(A_1 f_1^*>0)-I(A_1f_1>0)\big)\big] \notag\\
        =& \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_1f_1>0)\big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)\big] \notag\\
        &+ \mathbb{E} \big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2\tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)\big] \label{fclhs}
    \end{align}
    Therefore, by \eqref{firstterm}\eqref{secondterm}\eqref{fclhs}, we have 
    \begin{align}
        SV_\psi^* - SV_\psi(f_1,f_2;q) =& \text{First Term} + \text{Second Term} \notag\\
        \geq& \frac{C_\phi^2}{4}\mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(A_1f_1>0) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)\Big] \notag\\
        &+\frac{C^2_\phi}{2}\mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2\tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)\Big] \notag\\
        \geq& \frac{C_\phi^2}{4}\Big\{\mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2}) I(A_1f_1>0) \big(I(A_2\tilde f_2>0)-I(A_2f_2>0)\big)\Big] \notag \\
        &+ \mathbb{E} \Big[I(Y_2>q)q_{22}(\bar Z_{2}, \bar Y_{1}, \bar A_{2})I(A_2\tilde f_2>0)\big(I(A_1\tilde f_1>0)-I(A_1f_1>0)\big)\Big]\Big\} \notag\\
        =& \frac{C_\phi^2}{4}(SV^*-SV(f_1,f_2;q)),
    \end{align}
    which completes the proof.
\end{proof}





\subsection{Finite sample excess risk bound}
In this section, we discuss the risk bound with the surrogate form, i.e., the $\psi-$regret bound.
In fact, it suffices to bound the $\psi-$regret according to Fisher consistency.
Without loss of generality, we assume that the optimal $q^*$ has already been found, so that we can proceed with the following discussion for fixed $q = q^*$. 
To this end, we will re-clarify some notations that may be referred to in the following discussion:
\begin{align}
    & SV_\psi^{*} = SV_\psi(\tilde f_1,\tilde f_2;q^*) = \sup_{(f_1,f_2) \in \mathcal{F}}SV_\psi(f_1,f_2;q^*)
    = \sup_{(f_1,f_2) \in \mathcal{F}}\mathbb{P}\big[I(Y_2>q^*)\cdot q_{22}\cdot \psi(A_1f_1(H_1), A_2f_2(H_2)\big], \notag\\
\end{align}
\begin{align}
     (\tilde f_{n,1}, \tilde f_{n,2}) & = \argmax_{(f_1,f_2) \in \mathcal{U}_n} \big\{SV_\psi(f_1,f_2;q^*) - \lambda_n(||f_1||_{n,2}^2+||f_2||_{n,2}^2)\big\} \notag\\
     &= \argmax_{(f_1,f_2) \in \mathcal{U}_n}\big\{ \mathbb{P}\big[I(Y_2>q^*)\cdot q_{22}\cdot \psi(A_1f_1(H_1), A_2f_2(H_2)\big] - \lambda_n(||f_1||_{n,2}^2+||f_2||_{n,2}^2) \big\},\notag\\
     (\hat f_{n,1}, \hat f_{n,2}) & = \argmax_{(f_1,f_2) \in \mathcal{U}_n} \big\{\hat{SV}_\psi(f_1,f_2;q^*) - \lambda_n(||f_1||_{n,2}^2+||f_2||_{n,2}^2)\big\} \notag\\
     &= \argmax_{(f_1,f_2) \in \mathcal{U}_n} \big\{\mathbb{P}_n\big[I(Y_2>q^*)\cdot \hat q_{22}\cdot \psi(A_1f_1(H_1), A_2f_2(H_2)\big] - \lambda_n(||f_1||_{n,2}^2+||f_2||_{n,2}^2)\big\} \notag
\end{align}
where $\mathbb{P}, \mathbb{P}_n$ denote the true distribution and empirical distribution, respectively.
It is worth noting that the existence of $(\tilde f_{n,1},\tilde f_{n,2})$ is without any guarantee in general, but is intuitive in the following discussion. 

Based on (Laha2021), we have the following decomposition of $\psi-$regret:
\begin{align}
    &SV_\psi^* - SV_\psi(\hat f_{n,1}, \hat f_{n,2};q^*) \notag\\
    \leq& \quad \Big\{SV_\psi^* - [SV_\psi(\tilde f_{n,1}, \tilde f_{n,2};q^*) - \lambda_n(||\tilde f_{n,1}||_{n,2}^2+||\tilde f_{n,2}||_{n,2}^2)]\Big\} \notag\\
    &+ \Big\{SV_\psi(\tilde f_{n,1}, \tilde f_{n,2};q^*) -\lambda_n(||\tilde f_{n,1}||_{n,2}^2+||\tilde f_{n,2}||_{n,2}^2)\Big\} - \Big\{SV_\psi(\hat f_{n,1}, \hat f_{n,2};q^*) -\lambda_n(||\hat f_{n,1}||_{n,2}^2+||\hat f_{n,2}||_{n,2}^2)\Big\}
    \notag\\
    =&\quad \Big\{SV_\psi^* - SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) + \lambda_n(||\tilde f_{n,1}||_{n,2}^2+||\tilde f_{n,2}||_{n,2}^2) \Big\} \notag\\
     &+ \Big\{[SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) - \hat{SV}_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*)] - [SV_\psi(\hat f_{n,1},\hat f_{n,2};q^*) - \hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q^*)]\Big\} \notag\\
     &+ \Big\{[\hat{SV}_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) - \lambda_n(||\tilde f_{n,1}||_{n,2}^2+||\tilde f_{n,2}||_{n,2}^2)]- \sup_{(f_1,f_2)\in\mathcal{U}_n}[\hat{SV}_\psi(f_1,f_2;q^*) - \lambda_n(|| f_{n,1}||_{n,2}^2+||f_{n,2}||_{n,2}^2)] \Big\}\notag\\
     &+ \Big\{\sup_{(f_1,f_2)\in\mathcal{U}_n}[\hat{SV}_\psi(f_1,f_2;q^*) - \lambda_n(||f_{n,1}||_{n,2}^2+||f_{n,2}||_{n,2}^2)] - [\hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q^*)-\lambda_n(||\hat f_{n,1}||_{n,2}^2+||\hat f_{n,2}||_{n,2}^2)] \Big\} \notag\\
    \leq& \quad \Big\{SV_\psi^* - SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) + \lambda_n(||\tilde f_{n,1}||_{n,2}^2+||\tilde f_{n,2}||_{n,2}^2) \Big\} \notag\\
     &+ \Big\{[SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) - \hat{SV}_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*)] - [SV_\psi(\hat f_{n,1},\hat f_{n,2};q^*) - \hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q^*)]\Big\} \notag\\
     &+ \Big\{\sup_{(f_1,f_2)\in\mathcal{U}_n}[\hat{SV}_\psi(f_1,f_2;q^*) - \lambda_n(||f_{n,1}||_{n,2}^2+||f_{n,2}||_{n,2}^2)] - [\hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q^*)-\lambda_n(||\hat f_{n,1}||_{n,2}^2+||\hat f_{n,2}||_{n,2}^2)] \Big\}.
    \label{totalerror}
\end{align}

We first give a brief understanding of the three terms in \eqref{totalerror}.
\begin{itemize}
    \item The first term $\Big\{SV_\psi^* - SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) + \lambda_n(||\tilde f_{n,1}||_{n,2}^2+||\tilde f_{n,2}||_{n,2}^2) \Big\}$: it can be regarded as the approximation error since we restrict the function class of $f_1,f_2$ in $\mathcal{U}_n$. We denote it as $\textbf{App}_n$.
    
    \item The second term $\Big\{[SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*) - \hat{SV}_\psi(\tilde f_{n,1},\tilde f_{n,2};q^*)] - [SV_\psi(\hat f_{n,1},\hat f_{n,2};q^*) - \hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q^*)]\Big\}$, it is the main estimation error when we adopt empirical sample for practical estimation. 
    Similar to (Laha2021), we define the map 
    $\xi_{f_1,f_2,g_1,g_2}:\mathcal{H}_1\times\mathcal{H}_2\times\mathbb{R}^2\times\{-1,1\}^2 \to \mathbb{R}$,
    \begin{align*}
        \xi_{f_1,f_2,g_1,g_2}(\mathcal{D}):=& I(Y_2 > q^*) q_{22}\cdot \big[ \psi(A_1f_1(H_1),A_2f_2(H_2))- \psi(A_1g_1(H_1),A_2g_2(H_2))\big] \\
        =& I(Y_2 > q^*) q_{22} \cdot \big[ \phi(A_1f_1(H_1))\phi(A_2f_2(H_2))- \phi(A_1g_1(H_1))\phi(A_2g_2(H_2))\big].
    \end{align*}
    Then we can show that 
        \begin{align}
            & [SV_\psi(\tilde f_{n,1},\tilde f_{n,2};q) - SV_\psi(\hat f_{n,1},\hat f_{n,2};q)]-[\hat{SV}_\psi(\tilde f_{n,1},\tilde f_{n,2};q) -\hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q)] \notag\\
            =& \mathbb{P}\big\{I(Y_2>q^*) q_{22}\cdot[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2}) - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\big\} \notag\\
            &- \frac{1}{K} \sum_{k=1}^K \mathbb{P}_{n,k} \big\{I(Y_2>q^*) \hat q_{22}^{(-k)} \cdot [\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2})  - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\big\} \notag\\
            =& \mathbb{P}\big\{I(Y_2>q^*)q_{22}\cdot[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2}) - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\big\} - \mathbb{P}_n\big\{I(Y_2>q^*)q_{22}\cdot[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2}) - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\big\} \notag\\
            &+ \frac{1}{K} \sum_{k=1}^K \mathbb{P}_{n,k}\Big\{ \big\{I(Y_2>q^*)q_{22}\cdot[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2}) - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\big\}\notag\\
            &\quad - \big\{I(Y_2>q^*)\hat q_{22}^{(-k)}[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2})  - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\big\}\Big\} \notag\\
            =& (\mathbb{P}-\mathbb{P}_n)[\xi_{\tilde f_{n,1}, \tilde f_{n,2}, \hat f_{n,1}, \hat f_{n,2}}] + \frac{1}{K} \sum_{k=1}^K \mathbb{P}_{n,k}\Big\{I(Y_2>q^*)(q_{22}-\hat q_{22}^{(-k)})[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2})  - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\Big\} \notag\\
            \leq& (\mathbb{P}-\mathbb{P}_n)[\xi_{\tilde f_{n,1}, \tilde f_{n,2}, \hat f_{n,1}, \hat f_{n,2}}] + \sup_k \mathbb{P}_{n,k}\Big\{I(Y_2>q^*)(q_{22}-\hat q_{22}^{(-k)})[\psi(A_1\tilde f_{n,1},A_2 \tilde f_{n,2})  - \psi(A_1\hat f_{n,1},A_2 \hat f_{n,2})]\Big\}, \label{esterr}
        \end{align}
        where the second equality is by supposing the sample size $n$ is divisible by $K$ hence $\mathbb{P}_n = \frac{1}{K}\sum_{k=1}^K \mathbb{P}_{n,k}$.
        We denote the first term in \eqref{esterr} as $\textbf{Est}_n$ in line with (Laha2021), and the second term as $\textbf{Nuis}_n$ since it is caused by estimating the nuisance function. 

    \item The third term $\Big\{\sup\limits_{(f_1,f_2)\in\mathcal{U}_n}[\hat{SV}_\psi(f_1,f_2;q^*) - \lambda_n(||f_{n,1}||_{n,2}^2+||f_{n,2}||_{n,2}^2)] - [\hat{SV}_\psi(\hat f_{n,1},\hat f_{n,2};q^*)-\lambda_n(||\hat f_{n,1}||_{n,2}^2+||\hat f_{n,2}||_{n,2}^2)] \Big\}$: it is the optimization error that depends on the optimization method used to maximize penalized $\hat{SV}_\psi(f_1,f_2;q)$ over $\mathcal{U}_n$. We denote it as $\textbf{Opt}_n$.
\end{itemize}

Now we can summarize our $\psi-$risk bound as follows: 
\begin{align}
    SV_\psi^* - SV_\psi(\hat f_{n,1}, \hat f_{n,2};q) \leq \textbf{App}_n + \textbf{Est}_n + \textbf{Nuis}_n + \textbf{Opt}_n. \label{riskbound}
\end{align}
Under the above framework of error decomposition, we consider $\mathcal{U}_n$ be the class of neural networks, which is one of the most popular and effective nonparametric models when facing non-concave optimization.
To bound $\eqref{riskbound}$, we need the following assumptions.
\begin{ass}[]
    There exists a constant $C_m > 0$ such that $||q_{tt}||_\infty < C_m$ for $t=1,2$. \label{assbound1}
\end{ass}
Assumption \ref{assbound1} requires $q_{11}$ and $q_{22}$ are (uniformly) bounded, which is a standard assumption in DTR bounding.

\begin{ass}[]
    Suppose the nuisance function $\hat q_{tt}^{(-k)}$ is estimated from $k-$th fold in the cross fitting process. Then for any $\bar a_t \in \mathcal{A}_t, 1\leq k\leq K$, there exist constants $\lambda_t$ such that $||q_{tt}(\bar Z_t, \bar Y_{t-1}, \bar a_t) - \hat q_{tt}^{(-k)}(\bar Z_t, \bar Y_{t-1}, \bar a_t)||_{L_2(P)} = O(n^{-\gamma_t})$ for $t=1,2$. In addition, there exists a constant $C_{m'} > 0$ such that $\max\{||\hat q_{tt}^{(-k)}||_\infty\} < C_{m'}$ for $t=1,2$. \label{assbound2}
\end{ass}
Assumption \ref{assbound2} restricts the convergence rates of the estimated nuisance functions when the cross-fitting technique is employed and further requires a bound on them. 
Specifically, different estimation methods generally yield different convergence rates, and the estimation error of $q_{22}$ inevitably includes that of $q_{11}$.

\begin{ass}[Tsybakov small noise assumption]
    There exists a constant $C>0$, a small number $t_0\in(0,1)$, and positive reals $\alpha_1,\alpha_2>0$ such that
    \begin{align}
        P(0<|\eta_1(H_1)-1/2|\leq t) \leq Ct^{\alpha_1}, \quad P(0<|\eta_2(H_2)-1/2|\leq t) \leq Ct^{\alpha_1} \text{\quad for all\, } t\leq t_0,
    \end{align}
    where $\alpha$ is the Tsybakov noise exponents
    \label{assbound3.1}
\end{ass}
Moreover, since $q_{11}$ and $q_{22}$ are bounded, we have the following corollary.
\begin{cor}[]
Denote $\Delta_{1}(H_1) = SQ_1(H_1;1,q) - SQ_1(H_1;-1,q)$, $\Delta_{2}(H_2) = SQ_2(H_2;1,q) - SQ_2(H_2;-1,q)$. There exists a constant $C>0$, a small number $t_0\in(0,1)$, and a positive real $\alpha>0$ such that
    \begin{align}
        P(0<\Delta_1(H_1)\le t) + P(0<\Delta_2(H_2)\le t) \le Ct^\alpha, \text{\quad for all\, } t\leq t_0,
     \end{align}
    where $\alpha$ is the Tsybakov noise exponent.
    \label{assbound3.2}
\end{cor}
Assumptions \ref{assbound3.1} and \ref{assbound3.2} are the DTR formulation of Tsybakov small noise assumption, which is used to obtain a fast rate of convergence by controlling the distribution of the random variable near the decision boundary.

    generate 8 models corresponding to the following settings: $\tau$=0.5, $\phi$ types: 1,2,3,4, and model type: linear, neural network.
    For each model setting, run a Monte Carlo simulation of 100 repos and record the following indicators:

        \item standard deviation of above mean 
        \item mean of estimation error (defined as $QV(\tau,\hat {\bar d}_2) - \hat {QV}(\tau, \hat {\bar d}_2)$)
    \end{itemize}
    (Although we know the DGP, we still do not know the true distribution of each set of simulated data. Consequently, it is very difficult to obtain the true quantile values.)
    For training sample size $n$ = 500,1000,2000,5000, plot the values of true regret and RMSE (in two plots) of each model as $n$ increases.
    
    \item Comparative analysis:
    consider constructing two naive QTR estimators to compare and analyse the superiority of our method (prox\_qtr\_sl), the SRA, and the Oracle estimators.
    The SRA estimator is based on the sequential randomization assumption that omits the unmeasured confounders $U_t$, and the Oracle estimator is accessible to the true unmeasured confounders.
    The main difference between our methods under the proximal setting and SRA/Oracle is the weight term in \eqref{est}.
    Our methods adopt PIPW weights that require estimation of treatment bridge functions $q_{22}$, while SRA and Oracle concentrate only on the propensity score, but using different information about history covariates.
    For SRA estimator, denote $X_1^{SRA} = \{Y_0\}, X_2^{SRA} = \{\bar Y_1, A_1\}$, and we have the SRA weight
    \[
        \hat{W}^{SRA} := \frac{1}{\hat \pi_1(A_1|X_1^{SRA})\hat \pi_2(A_2|X_2^{SRA})}
        = \frac{1}{\hat \pi_1(A_1|Y_0)\hat \pi_2(A_2|\bar Y_1, A_1)}，
    \]
    where $\pi_1,\pi_2$ is the propensity score.
    Similarly, for Oracle estimator, denote $X_1^{Oracle} = \{Y_0,U_0\}, X_2^{Oracle}=\{\bar Y_1, \bar U_1, A_1\}$, and we have the Oracle weight
    \[
        \hat{W}^{Oracle} := \frac{1}{\hat \pi_1(A_1|X_1^{Oracle})\hat \pi_2(A_2|X_2^{Oracle})} 
        = \frac{1}{\hat \pi_1(A_1|Y_0,U_0)\hat \pi_2(A_2|\bar Y_1, \bar U_1, A_1)}.
    \]
    To get the SRA/Oracle estimator, we do the inner level optimization by
    \begin{align*}
        \hat Q_\tau^{SRA/Oracle}(Y_2(\bar d_2)) = \argmin_{q} \frac{1}{n}\sum_{i=1}^n \big\{W_i^{SRA/Oracle}I(d_1(Y_{0,i})=A_{1,i})I(d_2(\bar Y_{1,i},A_{1,i})=A_{2,i)}\cdot \rho_\tau(Y_{2,i}-q)\big\}.
    \end{align*}
    By the same outer level optimization, we have 
    \begin{align*}
        (\hat{f}_{1}^{SRA/Oracle}, \hat{f}_{2}^{SRA/Oracle}) = \argmax_{f_1, f_2 \in \mathcal{U}_n} \frac{1}{n}\sum_{i=1}^n \left[ I(Y_{2,i}>q) \cdot \hat{W}^{SRA/Oracle}_i \cdot \psi(A_{1,i}f_1(H_{1,i}), A_{2,i}f_2(H_{2,i})) \right].
    \end{align*}
    Run a Monte Carlo simulation of 100 repos and compare the following indicators of each model:
    \begin{itemize}
        \item mean of true quantile value estimated by the model (defined as $V(\hat {\bar d}_2)$)
        \item standard deviation of above mean 
        \item mean of estimated quantile value estimated by the model (defined as $\hat V(\hat {\bar d}_2)$)
    \end{itemize}
    $$ \text{Estimation Error} = \frac{1}{B} \sum_{i=1}^B \left| V(\hat{\bar{d}}_2^{(i)}) - \hat{V}(\hat{\bar{d}}_2^{(i)}) \right| $$
    
\end{itemize}
